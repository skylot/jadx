50 		int line = 0;
51 		int lastPos = 0;

53 		JResSearchNode lastNode = null;
54 		int searchStrLen = searchSettings.getSearchString().length();
55 		String content;
56 		try {
57 			content = resNode.getContent();
58 		} catch (Exception e) {
59 			LOG.error("Error load resource node content", e);
60 			return;
61 		}
62 		do {
63 			searchSettings.setStartPos(lastPos);
64 			int pos = searchSettings.find(content);
65 			if (pos > -1) {
66 				line += countLinesByPos(content, pos, lastPos);
67 				lastPos = pos + searchStrLen;
68 				String lineText = getLine(content, pos, lastPos);
69 				if (lastLineOccurred != line) {
70 					lastLineOccurred = line;
71 					if (lastNode != null) {
72 						emitter.onNext(lastNode);
73 					}
74 					lastNode = new JResSearchNode(resNode, lineText.trim(), line + 1, pos);
75 				}
76 			} else {
77 				if (lastNode != null) { // commit the final result node.
78 					emitter.onNext(lastNode);
79 				}
80 				break;
81 			}
82 		} while (!emitter.isCancelled() && lastPos < content.length());
